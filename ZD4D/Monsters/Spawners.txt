Class D4DMonsterSpawner : Actor 
{
	int user_Type;
	
	Default 
	{
		//Health 1;
		
		Monster;
		Health 2000000;
		//+NOINTERACTION
		+DONTCORPSE
		+NORADIUSDMG
		+NONSHOOTABLE
		-SHOOTABLE
		-COUNTKILL
		+BOSSDEATH
		+VULNERABLE
		+NOTIMEFREEZE
		+CANTSEEK
		+BOSSDEATH
		+NEVERTARGET
		+DONTTHRUST
		+DONTSPLASH
		+NOGRAVITY
		+NOPAIN
		+NOTRIGGER
		+NOTONAUTOMAP
		Species "D4DSpawner";
		DamageFactor "Normal", 0.0;
		DamageFactor "Massacre", 666;
		
	}
}




//===============================================================================
//===============================================================================
//	Zombieman
//===============================================================================
//===============================================================================

Class ZombieReplacer : D4DMonsterSpawner replaces Zombieman
{
	Default 
	{
		Radius 20;
		Height 56;
	}
	States
	{
	Spawn:
		//TNT1 A 0 NoDelay A_GiveInventory("NoDormantToken",1)
		TNT1 A 0 NODELAY
		{
			if(GetCvar("user_MonsterType") == 0)
			{
				A_SpawnItemEx("D4OldZombieman",0,0,0,0,0,0,0,MonsterSpawnFlags,0,tid);
			}
			if(GetCvar("user_MonsterType") == 1)
			{
				A_SpawnItemEx("D4Zombieman",0,0,0,0,0,0,0,MonsterSpawnFlags,0,tid);
			}
		}
		Goto End;

	//-----------------------------------------
	//	Special / End of spawner states
	//-----------------------------------------
	End:
		//TNT1 A 1 A_RadiusGive("DormantChecker",4,RGF_MONSTERS,1)
		TNT1 A 1
		{
			bTHRUACTORS = true;
			A_ChangeLinkFlags(true);
			A_Die();
		}
	Death:
		TNT1 A 1;
		Stop;
	}
}

				if(random(1,2)==1)
				{
					A_SpawnItemEx("D4FatsoTech",0,0,0,0,0,0,0,MonsterSpawnFlags,0,tid);
				}
				else
				{
					A_SpawnItemEx("D4FatsoHell",0,0,0,0,0,0,0,MonsterSpawnFlags,0,tid);

			}
		}
	Goto Teleport;

	//-----------------------------------------
	Teleport:
		TNT1 A 0 
		{
			bTHRUACTORS = true;
			A_ChangeLinkFlags(true);
		}
		
	Waiting:
		TNT1 A 70 A_JumpIf(IsPointerEqual(AAPTR_TRACER,AAPTR_NULL),"Death");
